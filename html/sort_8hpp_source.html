157 template<typename Scalar>
158 void InertiaSort<Scalar>::InertiaSortWithThreshold( InertiaSort<Scalar>* s )
159 {
160  // simplify method to find the values in the source SimpleMatrix<Scalar>
161  unsigned int start = s->m_sortOnColumns ?
162  s->m_sortRefIdx * s->m_data->getHeight() :
163  s->m_sortRefIdx;
164  unsigned int step = s->m_sortOnColumns ? 1 : s->m_data->getHeight();
165 
166  // copy values and compute total inertia
167  Scalar* values = s->m_data->getValues();
168  Scalar* result = s->m_sorted->getValues();
169  unsigned int* indexes = s->m_appendIdx ? s->m_indexes.get()->getValues() : nullptr;
170  Scalar totInertia = 0;
171  unsigned int nbValues = s->m_sortOnColumns ? s->m_data->getHeight() :
172  s->m_data->getWidth();
173  for (unsigned int i = 0; i < nbValues; ++i) {
174  result[i] = values[start+i*step];
175  totInertia += std::abs(values[start+i*step]);
176  }
177  s->m_totalInertia = totInertia;
178  if (s->m_appendIdx)
179  for( unsigned int i=0; i<nbValues; i++ )
180  indexes[i] = i;
181 
182  // extract values
183  Scalar inertia = 0;
184  unsigned int i = 0;
185  while (inertia < s->m_inertia * totInertia) {
186  unsigned int idxMax = i;
187  Scalar max = result[idxMax];
188  for ( unsigned int j = i; j < nbValues; j++ ) {
189  if (std::abs(result[j]) > max) {
190  idxMax = j;
191  max = std::abs(result[j]);
192  }
193  }
194  Scalar tmp = result[idxMax];
195  inertia += std::abs(tmp);
196  result[idxMax] = result[i];
197  result[i] = tmp;
198  if (s->m_appendIdx) {
199  unsigned int tmpIdx = indexes[i];
200  indexes[i] = indexes[idxMax];
201  indexes[idxMax] = tmpIdx;
202  }
203  ++i;
204  }
205  s->m_nbSortedValues = i;
206 }
