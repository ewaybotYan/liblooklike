188 template<typename Scalar>
189 void InertiaSort<Scalar>::InertiaSortWithThreshold( InertiaSort<Scalar>* s )
190 {
191  // simplify method to find the values in the source SimpleMatrix<Scalar>
192  unsigned int start = s->m_sortOnColumns ?
193  s->m_sortRefIdx * s->m_data->getHeight() :
194  s->m_sortRefIdx;
195  unsigned int step = s->m_sortOnColumns ? 1 : s->m_data->getHeight();
196 
197  // copy values and compute total inertia
198  Scalar* values = s->m_data->getValues();
199  Scalar* result = s->m_sorted->getValues();
200  unsigned int* indexes = s->m_appendIdx ? s->m_indexes.get()->getValues() : nullptr;
201  Scalar totInertia = 0;
202  unsigned int nbValues = s->m_sortOnColumns ? s->m_data->getHeight() :
203  s->m_data->getWidth();
204  for (unsigned int i = 0; i < nbValues; ++i) {
205  result[i] = values[start+i*step];
206  totInertia += std::abs(values[start+i*step]);
207  }
208  s->m_totalInertia = totInertia;
209  if (s->m_appendIdx)
210  for( unsigned int i=0; i<nbValues; i++ )
211  indexes[i] = i;
212 
213  // extract values
214  Scalar inertia = 0;
215  unsigned int i = 0;
216  while (inertia < s->m_inertia * totInertia) {
217  unsigned int idxMax = i;
218  Scalar max = result[idxMax];
219  for ( unsigned int j = i; j < nbValues; j++ ) {
220  if (std::abs(result[j]) > max) {
221  idxMax = j;
222  max = std::abs(result[j]);
223  }
224  }
225  Scalar tmp = result[idxMax];
226  inertia += std::abs(tmp);
227  result[idxMax] = result[i];
228  result[i] = tmp;
229  if (s->m_appendIdx) {
230  unsigned int tmpIdx = indexes[i];
231  indexes[i] = indexes[idxMax];
232  indexes[idxMax] = tmpIdx;
233  }
234  ++i;
235  }
236 
237  // resize indexes and sort array to the number of sorted values
238  if( s->m_appendIdx )
239  s->getSortIdx()->resizeVecHeight(i);
240  s->getSorted()->resizeVecHeight(i);
241 
242  s->m_nbSortedValues = i;
243 }
